class Stack { constructor() { this.top = null // 用于存储栈内元素的链表 this.size = 0 } // 进栈 push(val) { var node = { val: val, next: this.top, } this.top = node this.size++ return this } // 出栈 pop() { if (this.top) { var t = this.top.val this.top = this.top.next this.size-- return t } } // 查看栈顶元素的值，但不让它出栈 peek() { if (this.top) { return this.top.val } } size() { return this.size } }
